001    package listener;
002    
003    import org.apache.commons.logging.Log; 
004    import org.apache.commons.logging.LogFactory;
005    import javax.servlet.http.HttpSessionListener; 
006    import javax.servlet.http.HttpSessionEvent;
007    import org.codehaus.groovy.grails.commons.ApplicationHolder;
008    import serviceinterfaces.SessionCleanerServiceInt;
009    import org.springframework.context.ApplicationContext;
010    import java.util.HashSet;
011    
012    public class GDOCSessionListener implements HttpSessionListener { 
013            private static final Log log = LogFactory.getLog(GDOCSessionListener.class);
014    
015            public void sessionCreated(HttpSessionEvent event) { 
016                    log.debug("Session created"); 
017                    
018                    
019            }
020    
021            public void sessionDestroyed(HttpSessionEvent event) {
022                    log.debug("Session destroyed."); 
023                    String userId = (String) event.getSession().getAttribute("userId");
024                    HashSet tempLists = (HashSet) event.getSession().getAttribute("tempLists");
025                    HashSet tempAnalyses = (HashSet) event.getSession().getAttribute("tempAnalyses");
026                    ApplicationContext ctx = (ApplicationContext)ApplicationHolder.getApplication().getMainContext();
027                    SessionCleanerServiceInt service = (SessionCleanerServiceInt) ctx.getBean("cleanupService");
028                    if(service != null && userId != null && (tempLists !=null) && (tempAnalyses!=null)){
029                            log.debug(userId + " has destroyed session, cleanup temp data");
030                            if(!(tempLists.isEmpty()) || !(tempAnalyses.isEmpty())){
031                                    log.debug(userId + " found temporary data");
032                                    service.cleanup(userId,tempLists,tempAnalyses);
033                            }       
034                    }
035            }
036    }
